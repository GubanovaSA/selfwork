    "-- JOIN запрос для экспорта в CSV\n",
    "SELECT \n",
    "    c.name AS компания,\n",
    "    c.industry AS отрасль,\n",
    "    COUNT(DISTINCT cont.contact_id) AS количество_контактов,\n",
    "    COUNT(DISTINCT d.deal_id) AS количество_сделок,\n",
    "    COUNT(DISTINCT CASE WHEN d.stage = 'Закрытие выигрыш' THEN d.deal_id END) AS выигранные_сделки,\n",
    "    COUNT(DISTINCT CASE WHEN d.stage = 'Закрытие проигрыш' THEN d.deal_id END) AS проигранные_сделки,\n",
    "    ROUND(COUNT(DISTINCT CASE WHEN d.stage = 'Закрытие выигрыш' THEN d.deal_id END) * 100.0 / \n",
    "          NULLIF(COUNT(DISTINCT d.deal_id), 0), 2) AS процент_выигрыша,\n",
    "    COUNT(DISTINCT a.activity_id) AS количество_активностей\n",
    "FROM crm.companies c\n",
    "LEFT JOIN crm.contacts cont ON c.company_id = cont.company_id\n",
    "LEFT JOIN crm.deals d ON c.company_id = d.company_id\n",
    "LEFT JOIN crm.activities a ON cont.contact_id = a.contact_id\n",
    "GROUP BY c.name, c.industry\n",
    "ORDER BY выигранные_сделки DESC;"
   ]
