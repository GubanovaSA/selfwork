-- ============================================

-- Таблица 1: companies - хранение информации о компаниях-клиентах
CREATE TABLE IF NOT EXISTS companies (
    company_id SERIAL PRIMARY KEY,      -- Уникальный идентификатор компании
    name VARCHAR(200) NOT NULL,         -- Название компании
    industry VARCHAR(100),              -- Отрасль бизнеса
    website VARCHAR(100)                -- Сайт компании
);

-- Таблица 2: contacts - хранение контактных лиц компаний
CREATE TABLE IF NOT EXISTS contacts (
    contact_id SERIAL PRIMARY KEY,      -- Уникальный идентификатор контакта
    company_id INT REFERENCES companies(company_id), -- Связь с компанией
    first_name VARCHAR(50) NOT NULL,    -- Имя контакта
    last_name VARCHAR(50) NOT NULL,     -- Фамилия контакта
    email VARCHAR(100) UNIQUE           -- Email контакта (уникальный)
);

-- Таблица 3: deals - хранение информации о сделках
CREATE TABLE IF NOT EXISTS deals (
    deal_id SERIAL PRIMARY KEY,         -- Уникальный идентификатор сделки
    company_id INT REFERENCES companies(company_id), -- Связь с компанией
    deal_name VARCHAR(200) NOT NULL,    -- Название сделки
    stage VARCHAR(50),                  -- Стадия сделки (Квалификация/Предложение/Переговоры/Закрытие)
    amount NUMERIC(12,2)                -- Сумма сделки
);

-- Таблица 4: activities - хранение записей о взаимодействиях с клиентами
CREATE TABLE IF NOT EXISTS activities (
    activity_id SERIAL PRIMARY KEY,     -- Уникальный идентификатор активности
    contact_id INT REFERENCES contacts(contact_id), -- Связь с контактом
    type VARCHAR(50),                   -- Тип активности (Звонок/Email/Встреча и т.д.)
    notes TEXT,                         -- Заметки о взаимодействии
    activity_date DATE DEFAULT CURRENT_DATE -- Дата активности
);

-- Проверка: выводим список всех созданных таблиц
SELECT table_name 
FROM information_schema.tables 
WHERE table_schema = 'public' 
ORDER BY table_name;

-- ============================================
-- ЗАДАНИЕ 5: АНАЛИТИЧЕСКИЙ JOIN ЗАПРОС
-- ============================================

-- Аналитика эффективности работы с клиентами:
-- Собираем данные о компаниях, контактах, сделках и активностях
SELECT 
    c.name AS компания,                              -- Название компании
    c.industry AS отрасль,                           -- Отрасль бизнеса
    COUNT(DISTINCT cont.contact_id) AS количество_контактов,  -- Сколько контактов у компании
    COUNT(DISTINCT d.deal_id) AS количество_сделок,          -- Всего сделок
    COUNT(DISTINCT CASE WHEN d.stage = 'Закрытие выигрыш' THEN d.deal_id END) AS выигранные_сделки,  -- Успешные сделки
    COUNT(DISTINCT CASE WHEN d.stage = 'Закрытие проигрыш' THEN d.deal_id END) AS проигранные_сделки, -- Проигранные сделки
    ROUND(COUNT(DISTINCT CASE WHEN d.stage = 'Закрытие выигрыш' THEN d.deal_id END) * 100.0 / 
          NULLIF(COUNT(DISTINCT d.deal_id), 0), 2) AS процент_выигрыша,  -- Конверсия в успех
    COUNT(DISTINCT a.activity_id) AS количество_активностей,            -- Всего взаимодействий
    SUM(d.amount) AS общая_сумма_сделок,                                 -- Сумма всех сделок
    ROUND(AVG(d.amount), 2) AS средний_чек                               -- Средний чек по сделкам
FROM companies c
LEFT JOIN contacts cont ON c.company_id = cont.company_id
LEFT JOIN deals d ON c.company_id = d.company_id
LEFT JOIN activities a ON cont.contact_id = a.contact_id
GROUP BY c.name, c.industry
ORDER BY выигранные_сделки DESC, общая_сумма_сделок DESC;
-- Результат: аналитическая таблица для экспорта в CSV и визуализации


-- ============================================
-- ДОПОЛНИТЕЛЬНАЯ АНАЛИТИКА: Распределение по отраслям
-- ============================================

-- Сводная статистика по отраслям: сколько компаний, сделок и сумма
SELECT 
    c.industry,
    COUNT(DISTINCT c.company_id) AS companies,
    COUNT(d.deal_id) AS deals,
    SUM(d.amount) AS total_amount,
    ROUND(AVG(d.amount), 2) AS avg_amount
FROM companies c
LEFT JOIN deals d ON c.company_id = d.company_id
GROUP BY c.industry
ORDER BY total_amount DESC;
-- Результат: помогает определить самые прибыльные отрасли
